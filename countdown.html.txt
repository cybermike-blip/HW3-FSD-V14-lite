    <script>
        // Tailwind script already loaded
        function initTailwind() {
            tailwind.config = {
                content: [],
                theme: {
                    extend: {}
                }
            }
        }

        // Countdown logic
        function startCountdown() {
            // Target: June 30, 2026 23:59:59 PDT (end of June)
            const targetDate = new Date("2026-06-30T23:59:59-07:00").getTime();
            
            const countdownEl = document.getElementById("countdown");
            
            function updateCountdown() {
                const now = Date.now();
                const distance = targetDate - now;
                
                if (distance < 0) {
                    countdownEl.innerHTML = `
                        <div class="col-span-4 text-center py-8">
                            <div class="text-5xl font-bold text-[#e82127]">FSD v14 LITE IS HERE!</div>
                            <p class="mt-4 text-xl">Check your Tesla app for the update</p>
                        </div>
                    `;
                    return;
                }
                
                const days = Math.floor(distance / (1000 * 60 * 60 * 24));
                const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((distance % (1000 * 60)) / 1000);
                
                countdownEl.innerHTML = `
                    <div class="countdown-digit rounded-3xl p-6 text-center">
                        <div class="text-7xl font-bold text-white">${days}</div>
                        <div class="text-xs uppercase tracking-[2px] mt-2 text-white/50">DAYS</div>
                    </div>
                    <div class="countdown-digit rounded-3xl p-6 text-center">
                        <div class="text-7xl font-bold text-white">${hours}</div>
                        <div class="text-xs uppercase tracking-[2px] mt-2 text-white/50">HOURS</div>
                    </div>
                    <div class="countdown-digit rounded-3xl p-6 text-center">
                        <div class="text-7xl font-bold text-white">${minutes}</div>
                        <div class="text-xs uppercase tracking-[2px] mt-2 text-white/50">MINUTES</div>
                    </div>
                    <div class="countdown-digit rounded-3xl p-6 text-center">
                        <div class="text-7xl font-bold text-white">${seconds}</div>
                        <div class="text-xs uppercase tracking-[2px] mt-2 text-white/50">SECONDS</div>
                    </div>
                `;
            }
            
            updateCountdown();
            setInterval(updateCountdown, 1000);
        }

        // Share to X
        function shareToX() {
            const text = encodeURIComponent("HW3 Tesla owners — FSD v14 Lite drops end of June! Countdown live 👇");
            const url = encodeURIComponent(window.location.href);
            window.open(`https://x.com/intent/tweet?text=${text}&url=${url}`, "_blank");
        }

        // Initialize everything
        window.onload = function() {
            initTailwind();
            startCountdown();
            
            // Keyboard shortcut easter egg
            console.log('%c🚀 HW3 FSD v14 Lite countdown ready. Press "C" to copy embed code.', 'color:#e82127; font-weight:bold');
            document.addEventListener('keydown', (e) => {
                if (e.key === 'c' || e.key === 'C') {
                    navigator.clipboard.writeText('<iframe src="' + window.location.href + '" width="100%" height="800" frameborder="0"></iframe>');
                    alert("✅ Embed code copied to clipboard!");
                }
            });
        };
